22 "Comma-separated list of actor names to publish object lists for", false, false, true);

37template <typename T>
38void CarlaConverter::declareAndLoadParameter(const std::string& name,
39 T& param,
40 const std::string& description,
41 const bool add_to_auto_reconfigurable_params,
42 const bool is_required,
43 const bool read_only,
44 const std::optional<double>& from_value,
45 const std::optional<double>& to_value,
46 const std::optional<double>& step_value,
47 const std::string& additional_constraints) {
48 rcl_interfaces::msg::ParameterDescriptor param_desc;
49 param_desc.description = description;
50 param_desc.additional_constraints = additional_constraints;
51 param_desc.read_only = read_only;
52
53 auto type = rclcpp::ParameterValue(param).get_type();
54
55 if (from_value.has_value() && to_value.has_value()) {
56 if constexpr (std::is_integral_v<T>) {
57 rcl_interfaces::msg::IntegerRange range;
58 range.set__from_value(static_cast<T>(from_value.value())).set__to_value(static_cast<T>(to_value.value()));
59 if (step_value.has_value()) range.set__step(static_cast<T>(step_value.value()));
60 param_desc.integer_range = {range};
61 } else if constexpr (std::is_floating_point_v<T>) {
62 rcl_interfaces::msg::FloatingPointRange range;
63 range.set__from_value(static_cast<T>(from_value.value())).set__to_value(static_cast<T>(to_value.value()));
64 if (step_value.has_value()) range.set__step(static_cast<T>(step_value.value()));
65 param_desc.floating_point_range = {range};
66 } else {
67 RCLCPP_WARN(this->get_logger(), "Parameter type of parameter '%s' does not support specifying a range", name.c_str());
68 }
69 }
70
71 this->declare_parameter(name, type, param_desc);
72
73 try {
74 param = this->get_parameter(name).get_value<T>();
75 std::stringstream ss;
76 ss << "Loaded parameter '" << name << "': ";
77 if constexpr (is_vector_v<T>) {
78 ss << "[";
79 for (const auto& element : param) ss << element << (&element != &param.back() ? ", " : "");
80 ss << "]";
81 } else {
82 ss << param;
83 }
84 RCLCPP_INFO_STREAM(this->get_logger(), ss.str());
85 } catch (rclcpp::exceptions::ParameterUninitializedException&) {
86 if (is_required) {
87 RCLCPP_FATAL_STREAM(this->get_logger(), "Missing required parameter '" << name << "', exiting");
88 exit(EXIT_FAILURE);
89 } else {
90 std::stringstream ss;
91 ss << "Missing parameter '" << name << "', using default value: ";
92 if constexpr (is_vector_v<T>) {
93 ss << "[";
94 for (const auto& element : param) ss << element << (&element != &param.back() ? ", " : "");
95 ss << "]";
96 } else {
97 ss << param;
98 }
99 RCLCPP_WARN_STREAM(this->get_logger(), ss.str());
100 this->set_parameters({rclcpp::Parameter(name, rclcpp::ParameterValue(param))});
101 }
102 }
103
104 if (add_to_auto_reconfigurable_params) {
105 std::function<void(const rclcpp::Parameter&)> setter = [&param](const rclcpp::Parameter& p) { param = p.get_value<T>(); };
106 auto_reconfigurable_params_.push_back(std::make_tuple(name, setter));
107 }
108}

110rcl_interfaces::msg::SetParametersResult CarlaConverter::parametersCallback(const std::vector<rclcpp::Parameter>& parameters) {
111 for (const auto& param : parameters) {
112 for (auto& auto_reconfigurable_param : auto_reconfigurable_params_) {
113 if (param.get_name() == std::get<0>(auto_reconfigurable_param)) {
114 std::get<1>(auto_reconfigurable_param)(param);
115 RCLCPP_INFO(this->get_logger(), "Reconfigured parameter '%s' to: %s", param.get_name().c_str(),
116 param.value_to_string().c_str());
117 break;
118 }
119 }
120 }
121
122 rcl_interfaces::msg::SetParametersResult result;
123 result.successful = true;
124
125 return result;
126}

128void CarlaConverter::setup() {
129 // callback for dynamic parameter configuration
130 parameters_callback_ =
131 this->add_on_set_parameters_callback(std::bind(&CarlaConverter::parametersCallback, this, std::placeholders::_1));
132
133 // parse comma-separated actor name strings into vectors
134 std::string actor_name;
135 std::stringstream ego_data_actors_ss(ego_data_actors_string_);
136 while (std::getline(ego_data_actors_ss, actor_name, ',')) {
137 actor_name.erase(std::remove_if(actor_name.begin(), actor_name.end(), isspace), actor_name.end());
138 ego_data_actors_.push_back(actor_name);
139 }
140 std::stringstream object_list_actors_ss(object_list_actors_string_);
141 while (std::getline(object_list_actors_ss, actor_name, ',')) {
142 actor_name.erase(std::remove_if(actor_name.begin(), actor_name.end(), isspace), actor_name.end());
143 object_list_actors_.push_back(actor_name);
144 }
145
146 // setup callback functions
147 auto odometryArgCallback = [this](const std::string& actor_name) {
148 return
149 [this, actor_name](const nm::Odometry::ConstSharedPtr msg) -> void { CarlaConverter::odometryCallback(msg, actor_name); };
150 };
151
152 auto vehicleStatusArgCallback = [this](const std::string& actor_name) {
153 return [this, actor_name](const cm::CarlaEgoVehicleStatus::ConstSharedPtr msg) -> void {
154 CarlaConverter::vehicleStatusCallback(msg, actor_name);
155 };
156 };
157
158 auto vehicleInfoArgCallback = [this](const std::string& actor_name) {
159 return [this, actor_name](const cm::CarlaEgoVehicleInfo::ConstSharedPtr msg) -> void {
160 CarlaConverter::vehicleInfoCallback(msg, actor_name);
161 };
162 };
163
164 auto gnssArgCallback = [this](const std::string& actor_name) {
165 return
166 [this, actor_name](const ssm::NavSatFix::ConstSharedPtr msg) -> void { CarlaConverter::gnssCallback(msg, actor_name); };
167 };
168
169 auto imuArgCallback = [this](const std::string& actor_name) {
170 return [this, actor_name](const ssm::Imu::ConstSharedPtr msg) -> void { CarlaConverter::imuCallback(msg, actor_name); };
171 };
172
173 // setup tf2 buffer
174 tf2_buffer_ = std::make_unique<tf2_ros::Buffer>(this->get_clock());
175 tf2_buffer_->setUsingDedicatedThread(true);
176 tf2_listener_ = std::make_shared<tf2_ros::TransformListener>(*tf2_buffer_);
177
178 // setup qos
179 rclcpp::QoS qosLatching = rclcpp::QoS(rclcpp::KeepLast(1));
180 qosLatching.transient_local();
181 qosLatching.reliable();
182
183 // setup subscriber and publisher
184 sub_objects_ = this->create_subscription<dom::ObjectArray>(
185 kCarlaObjectsTopic, 1, std::bind(&CarlaConverter::objectsCallback, this, std::placeholders::_1));
186 pub_objects_carla_map_ = this->create_publisher<pi::ObjectList>("~/object_list", 1);
187 RCLCPP_INFO(this->get_logger(), "Subscribed to '%s'", sub_objects_->get_topic_name());
188 RCLCPP_INFO(this->get_logger(), "Publishing to '%s'", pub_objects_carla_map_->get_topic_name());
189
190 if (enable_traffic_lights_) {
191 sub_traffic_light_info_ = this->create_subscription<cm::CarlaTrafficLightInfoList>(
192 kCarlaTrafficLightsInfoTopic, 1, std::bind(&CarlaConverter::trafficLightInfoCallback, this, std::placeholders::_1));
193 sub_traffic_light_status_ = this->create_subscription<cm::CarlaTrafficLightStatusList>(
194 kCarlaTrafficLightsStatusTopic, 1, std::bind(&CarlaConverter::trafficLightStatusCallback, this, std::placeholders::_1));
195 pub_traffic_lights_carla_map_ = this->create_publisher<pi::ObjectList>("~/traffic_lights", 1);
196 timer_traffic_lights_ = create_wall_timer(std::chrono::milliseconds(static_cast<int>(1000 / traffic_light_frequency_)),
197 std::bind(&CarlaConverter::publishTrafficLights, this));
198 RCLCPP_INFO(this->get_logger(), "Subscribed to '%s' and '%s'", sub_traffic_light_info_->get_topic_name(),
199 sub_traffic_light_status_->get_topic_name());
200 RCLCPP_INFO(this->get_logger(), "Publishing to '%s'", pub_traffic_lights_carla_map_->get_topic_name());
201 } else {
202 RCLCPP_INFO(this->get_logger(), "Traffic light subscriptions and publishing are disabled.");
203 }
204
205 sub_world_info_ = this->create_subscription<cm::CarlaWorldInfo>(
206 kCarlaWorldInfoTopic, qosLatching, std::bind(&CarlaConverter::worldInfoCallback, this, std::placeholders::_1));
207 pub_map_info_ = this->create_publisher<stm::String>("~/map_info", qosLatching);
208 RCLCPP_INFO(this->get_logger(), "Subscribed to '%s'", sub_world_info_->get_topic_name());
209 RCLCPP_INFO(this->get_logger(), "Publishing to '%s'", pub_map_info_->get_topic_name());
210
211 // setup subscriber and publisher depending on actor_name
212 for (std::string& actor_name : ego_data_actors_) {
213 Subscriber<nm::Odometry> sub_odometry =
214 this->create_subscription<nm::Odometry>("/carla/" + actor_name + "/odometry", 1, odometryArgCallback(actor_name));
215 Subscriber<cm::CarlaEgoVehicleStatus> sub_vehicle_status = this->create_subscription<cm::CarlaEgoVehicleStatus>(
216 "/carla/" + actor_name + "/vehicle_status", 1, vehicleStatusArgCallback(actor_name));
217 Subscriber<cm::CarlaEgoVehicleInfo> sub_vehicle_info = this->create_subscription<cm::CarlaEgoVehicleInfo>(
218 "/carla/" + actor_name + "/vehicle_info", qosLatching, vehicleInfoArgCallback(actor_name));
219 Subscriber<ssm::NavSatFix> sub_gnss =
220 this->create_subscription<ssm::NavSatFix>("/carla/" + actor_name + "/gnss", 1, gnssArgCallback(actor_name));
221 Subscriber<ssm::Imu> sub_imu =
222 this->create_subscription<ssm::Imu>("/carla/" + actor_name + "/imu", 1, imuArgCallback(actor_name));
223
224 // save subscriber in map with actor_name as key
225 sub_odometry_map_.insert({actor_name, sub_odometry});
226 sub_vehicle_status_map_.insert({actor_name, sub_vehicle_status});
227 sub_vehicle_info_map_.insert({actor_name, sub_vehicle_info});
228 sub_gnss_map_.insert({actor_name, sub_gnss});
229 sub_imu_map_.insert({actor_name, sub_imu});
230
231 // setup publisher depending on actor_name
232 Publisher<pi::EgoData> pub_ego_data = this->create_publisher<pi::EgoData>("~/" + actor_name + "/ego_data", 1);
233
234 // save publisher in map with actor_name as key
235 pub_ego_data_map_.insert({actor_name, pub_ego_data});
236
237 ROS_LOG_STREAM(
238 INFO, "Subscribed /carla state topics for actor " << actor_name << " and publishing ~/" << actor_name << "/ego_data");
239 }
240
241 // setup object subscriber and publisher for object_list_actors_
242 for (std::string& actor_name : object_list_actors_) {
243 // setup subscriber depending on actor_name
244 Subscriber<cm::CarlaEgoVehicleInfo> sub_vehicle_info = this->create_subscription<cm::CarlaEgoVehicleInfo>(
245 "/carla/" + actor_name + "/vehicle_info", qosLatching, vehicleInfoArgCallback(actor_name));
246
247 // save subscriber in map with actor_name as key
248 sub_vehicle_info_map_.insert({actor_name, sub_vehicle_info});
249
250 // setup publisher depending on actor_name
251 Publisher<pi::ObjectList> pub_objects = this->create_publisher<pi::ObjectList>("~/" + actor_name + "/object_list", 1);
252
253 // save publisher in map with actor_name as key
254 pub_objects_map_.insert({actor_name, pub_objects});
255
256 ROS_LOG_STREAM(
257 INFO, "Subscribed /carla object topics for actor " << actor_name << " and publishing ~/" << actor_name << "/object_list");
258 }
259
260 // create 1s timer to subscribe to custom topics
261 timer_ = this->create_wall_timer(std::chrono::seconds(1), std::bind(&CarlaConverter::subscribeCustomTopics, this));
262
263 RCLCPP_INFO(this->get_logger(), "carla_converter running...");
264}

266void CarlaConverter::subscribeCustomTopics() {
267 auto customObjectsArgCallback = [this](const std::string& topic_name) {
268 return [this, topic_name](const dom::ObjectArray::ConstSharedPtr msg) -> void {
269 CarlaConverter::customObjectsCallback(msg, topic_name);
270 };
271 };
272 // iterate over all topics
273 std::map<std::string, std::vector<std::string>> topics = this->get_topic_names_and_types();
274 for (const auto& topic : topics) {
275 std::string topic_name = topic.first;
276 std::string topic_type = topic.second[0];
277
278 // skip if topic does not match type
279 if (topic_type != "derived_object_msgs/msg/ObjectArray") continue;
280
281 // skip if topic matches standard object topic pattern
282 std::regex pattern_objects("/carla/.*\\/objects");
283 if (topic_name == kCarlaObjectsTopic || std::regex_match(topic_name, pattern_objects)) continue;
284
285 // remove "/carla/"
286 size_t pos = topic_name.find("/carla/");
287 if (pos != std::string::npos) {
288 topic_name.erase(pos, 7);
289 }
290
291 // skip if topic is already subscribed
292 if (sub_custom_objects_map_.find(topic_name) != sub_custom_objects_map_.end()) continue;
293
294 Subscriber<dom::ObjectArray> sub_custom_objects =
295 this->create_subscription<dom::ObjectArray>("/carla/" + topic_name, 1, customObjectsArgCallback(topic_name));
296 sub_custom_objects_map_.insert({topic_name, sub_custom_objects});
297
298 Publisher<pi::ObjectList> pub_custom_objects = this->create_publisher<pi::ObjectList>("~/" + topic_name, 1);
299 pub_custom_objects_map_.insert({topic_name, pub_custom_objects});
300
301 ROS_LOG_STREAM(INFO, "Subscribed custom topic /carla/" << topic_name << " and publishing ~/" << topic_name);
302 }
303}

333void CarlaConverter::trafficLightInfoCallback(const cm::CarlaTrafficLightInfoList::ConstSharedPtr msg) {
334 try {
335 msg_traffic_lights_ = std::make_shared<pi::ObjectList>();
336
337 for (auto& traffic_light : msg->traffic_lights) {
338 pi::Object pi_light;
339 pi_light.id = traffic_light.id;
340 pi_light.existence_probability = 1.0;
341 oa::initializeState(pi_light, pi::TRAFFICLIGHT::MODEL_ID);
342 oa::setPose(pi_light.state, traffic_light.transform);
343 oa::setTrafficLightType(pi_light.state, pi::TRAFFICLIGHT::TYPE_UNKNOWN);
344 oa::setTrafficLightState(pi_light.state, pi::TRAFFICLIGHT::STATE_UNKNOWN);
345
346 msg_traffic_lights_->objects.push_back(pi_light);
347 }
348 } catch (const std::exception& e) {
349 RCLCPP_WARN(this->get_logger(), "Skip TrafficLightInfo processing: %s", e.what());
350 }

359int convert_traffic_status(const int status_carla) {
360 switch (status_carla) {
361 case carla_msgs::msg::CarlaTrafficLightStatus::RED:
362 return pi::TRAFFICLIGHT::STATE_RED;
363 case carla_msgs::msg::CarlaTrafficLightStatus::YELLOW:
364 return pi::TRAFFICLIGHT::STATE_YELLOW;
365 case carla_msgs::msg::CarlaTrafficLightStatus::GREEN:
366 return pi::TRAFFICLIGHT::STATE_GREEN;
367 case carla_msgs::msg::CarlaTrafficLightStatus::UNKNOWN:
368 return pi::TRAFFICLIGHT::STATE_UNKNOWN;
369 default:
370 return pi::TRAFFICLIGHT::STATE_UNKNOWN;
371 }
372}

374void CarlaConverter::trafficLightStatusCallback(const cm::CarlaTrafficLightStatusList::ConstSharedPtr msg) {
375 try {
376 if (!msg_traffic_lights_ || msg_traffic_lights_->objects.empty()) {
377 RCLCPP_WARN(this->get_logger(), "No traffic lights available to process status.");
378 return;
379 }
380
381 for (const auto carla_traffic_light : msg->traffic_lights) {
382 for (auto& pi_traffic_light : msg_traffic_lights_->objects) {
383 if (pi_traffic_light.id == carla_traffic_light.id) {
384 oa::setTrafficLightState(pi_traffic_light.state, convert_traffic_status(carla_traffic_light.state));
385 break;
386 }
387 }
388 }
389 } catch (const std::exception& e) {
390 RCLCPP_WARN(this->get_logger(), "Skip TrafficLightStatus processing: %s", e.what());
391 }
392}

394void CarlaConverter::worldInfoCallback(const cm::CarlaWorldInfo::ConstSharedPtr msg) {
395 const std::string current_map_name = msg->map_name;
396 std::smatch match;
397 std::string carla_map_name;
398
399 // check if the string matches the default pattern
400 std::regex pattern_default_map(R"(Carla/Maps/([^/]+))");
401 if (std::regex_match(current_map_name, match, pattern_default_map)) {
402 carla_map_name = match[1];
403 }
404
405 // check if the string matches the custom pattern
406 std::regex pattern_custom_map(R"((.+)/Maps/([^/]+)/\2)");
407 if (std::regex_match(current_map_name, match, pattern_custom_map)) {
408 carla_map_name = match[2];
409 }
410
411 if (carla_map_name.empty()) {
412 RCLCPP_ERROR(this->get_logger(), "Wrong format of CARLA map name: '%s'", current_map_name.c_str());
413 return;
414 }
415
416 RCLCPP_INFO(this->get_logger(), "Received world info for map '%s'.", carla_map_name.c_str());
417
418 stm::String map_info_msg;
419 map_info_msg.data = carla_map_name;
420 pub_map_info_->publish(map_info_msg);
421}

433void CarlaConverter::odometryCallback(const nm::Odometry::ConstSharedPtr msg, std::string actor_name) {
434 // map ego data from CARLA to the perception_msgs EgoData format
435 if (ego_shape_set_map_[actor_name] && ego_status_set_map_[actor_name]) {
436 msg_ego_data_.header = msg->header;
437
438 // get euler angles
439 double roll = 0.0;
440 double pitch = 0.0;
441 double yaw = 0.0;
442 tf2::Quaternion quat_tf;
443 tf2::fromMsg(msg->pose.pose.orientation, quat_tf);
444 tf2::Matrix3x3 matrix(quat_tf);
445 matrix.getRPY(roll, pitch, yaw);
446
447 // fill state
448 oa::initializeState(msg_ego_data_, pi::EGO::MODEL_ID);
449 msg_ego_data_.state.header = msg_ego_data_.header;
450 oa::setPose(msg_ego_data_.state, msg->pose.pose);
451 oa::setZ(msg_ego_data_, msg->pose.pose.position.z + ego_shape_map_[actor_name].dimensions[2] / 2.0);
452
453 oa::setYawRate(msg_ego_data_.state,
454 msg->twist.twist.angular.z); // twist is defined in child frame (no transformation needed)
455 oa::setVelocity(msg_ego_data_.state, msg->twist.twist.linear); // twist is defined in child frame (no transformation needed)
456 oa::setAcceleration(msg_ego_data_.state, ego_acceleration_map_[actor_name].linear);
457
458 // SteeringAngleMax is given in rad (contrary to the description in the documentation)
459 // https://github.com/carla-simulator/ros-bridge/blob/e9063d97ff5a724f76adbb1b852dc71da1dcfeec/carla_ros_bridge/src/carla_ros_bridge/ego_vehicle.py#L145C46-L145C81
460 oa::setSteeringAngleAck(msg_ego_data_.state, -ego_steering_angle_map_[actor_name] * ego_steering_angle_max_map_[actor_name]);
461 oa::setStandstill(msg_ego_data_.state, std::sqrt(pow(msg->twist.twist.linear.x, 2) + pow(msg->twist.twist.linear.y, 2) +
462 pow(msg->twist.twist.linear.z, 2)) <= 0.01);
463
464 // reference point for object position
465 msg_ego_data_.state.reference_point.value = pi::ObjectReferencePoint::GEOMETRIC_CENTER;
466
467 // fill variances
468 oa::setContinuousStateCovarianceAt(msg_ego_data_.state, pi::EGO::X, pi::EGO::X, pos_variances_);
469 oa::setContinuousStateCovarianceAt(msg_ego_data_.state, pi::EGO::Y, pi::EGO::Y, pos_variances_);
470 oa::setContinuousStateCovarianceAt(msg_ego_data_.state, pi::EGO::Z, pi::EGO::Z, pos_variances_);
471 oa::setContinuousStateCovarianceAt(msg_ego_data_.state, pi::EGO::VEL_LON, pi::EGO::VEL_LON, vel_variances_);
472 oa::setContinuousStateCovarianceAt(msg_ego_data_.state, pi::EGO::VEL_LAT, pi::EGO::VEL_LAT, vel_variances_);
473 oa::setContinuousStateCovarianceAt(msg_ego_data_.state, pi::EGO::ACC_LON, pi::EGO::ACC_LON, acc_variances_);
474 oa::setContinuousStateCovarianceAt(msg_ego_data_.state, pi::EGO::ACC_LAT, pi::EGO::ACC_LAT, acc_variances_);
475 oa::setContinuousStateCovarianceAt(msg_ego_data_.state, pi::EGO::ROLL, pi::EGO::ROLL, angle_variances_);
476 oa::setContinuousStateCovarianceAt(msg_ego_data_.state, pi::EGO::PITCH, pi::EGO::PITCH, angle_variances_);
477 oa::setContinuousStateCovarianceAt(msg_ego_data_.state, pi::EGO::YAW, pi::EGO::YAW, angle_variances_);
478 oa::setContinuousStateCovarianceAt(msg_ego_data_.state, pi::EGO::YAW_RATE, pi::EGO::YAW_RATE, angle_rate_variances_);
479
480 // set vehicle id and dimensions
481 msg_ego_data_.vehicle_id = ego_id_map_[actor_name];

504void CarlaConverter::objectsCallback(const dom::ObjectArray::ConstSharedPtr msg) {
505 pi::ObjectList msg_object_list_ = CarlaConverter::convertObjectArray(msg);
506
507 // publish object_list in carla_map frame
508 pub_objects_carla_map_->publish(msg_object_list_);
509
510 // iterate over each actor_name
511 for (std::string& actor_name : object_list_actors_) {
512 pi::ObjectList msg_object_list_copy = msg_object_list_;
513
514 // Set increasing sensor ID per role name. As `ego_data_actors_` doesnt change, this assigns a fixed sensor ID to each role.
515 for (auto& object : msg_object_list_copy.objects) {
516 object.state.sensor_id[0] = ego_id_map_[actor_name];
517 }
518
519 // remove element with the actor_name id
520 msg_object_list_copy.objects.erase(
521 std::remove_if(msg_object_list_copy.objects.begin(), msg_object_list_copy.objects.end(),
522 [this, actor_name](const pi::Object& obj) { return obj.id == ego_id_map_[actor_name]; }),
523 msg_object_list_copy.objects.end());
524
525 // transform the object_list from carla_map to actor_name frame
526 pi::ObjectList msg_object_list_transformed;
527 if (CarlaConverter::transformFrame(msg_object_list_copy, msg_object_list_transformed, actor_name)) {
528 pub_objects_map_[actor_name]->publish(msg_object_list_transformed);
529 }

545pi::ObjectList CarlaConverter::convertObjectArray(const dom::ObjectArray::ConstSharedPtr msg) {
546 // map the objects from the CARLA format to the perception_msgs format
547 pi::ObjectList msg_object_list_;
548 msg_object_list_.header = msg->header;
549
550 for (size_t i = 0; i < msg->objects.size(); i++) {
551 // only add classified objects
552 if (!msg->objects[i].object_classified) continue;
553
554 for (auto& actor_name : ego_data_actors_) {
555 // only continue if ego_info_set_
556 if (!ego_info_set_map_[actor_name]) continue;
557
558 // get shape from actor_name vehicle
559 if (ego_id_map_[actor_name] == msg->objects[i].id) {
560 ego_shape_map_[actor_name] = msg->objects[i].shape;
561 ego_shape_set_map_[actor_name] = true;
562 }
563 }
564
565 pi::Object objectTemp;
566 objectTemp.id = msg->objects[i].id;
567 objectTemp.existence_probability = 1.0; // always 1.0 as source is CARLA
568
569 // get yaw angle
570 double roll = 0.0;
571 double pitch = 0.0;
572 double yaw = 0.0;
573 tf2::Quaternion quat_tf;
574 tf2::fromMsg(msg->objects[i].pose.orientation, quat_tf);
575 tf2::Matrix3x3 matrix(quat_tf);
576 matrix.getRPY(roll, pitch, yaw);
577
578 // fill state
579 oa::initializeState(objectTemp, pi::HEXAMOTION::MODEL_ID);
580 objectTemp.state.header = msg->objects[i].header;
581 oa::setPose(objectTemp.state, msg->objects[i].pose);
582 oa::setVelocityXYZYaw(objectTemp.state, msg->objects[i].twist.linear, yaw);
583 oa::setAccelerationXYZYaw(objectTemp.state, msg->objects[i].accel.linear, yaw);
584 oa::setRollRate(objectTemp.state, msg->objects[i].twist.angular.x);
585 oa::setPitchRate(objectTemp.state, msg->objects[i].twist.angular.y);
586 oa::setYawRate(objectTemp.state, msg->objects[i].twist.angular.z);
587 oa::setLength(objectTemp, msg->objects[i].shape.dimensions[0]);
588 oa::setWidth(objectTemp, msg->objects[i].shape.dimensions[1]);
589 oa::setHeight(objectTemp, msg->objects[i].shape.dimensions[2]);
590
591 // fill variances
592 oa::setContinuousStateCovarianceAt(objectTemp, pi::HEXAMOTION::X, pi::HEXAMOTION::X, pos_variances_);
593 oa::setContinuousStateCovarianceAt(objectTemp, pi::HEXAMOTION::Y, pi::HEXAMOTION::Y, pos_variances_);
594 oa::setContinuousStateCovarianceAt(objectTemp, pi::HEXAMOTION::Z, pi::HEXAMOTION::Z, pos_variances_);
595 oa::setContinuousStateCovarianceAt(objectTemp, pi::HEXAMOTION::VEL_LON, pi::HEXAMOTION::VEL_LON, vel_variances_);
596 oa::setContinuousStateCovarianceAt(objectTemp, pi::HEXAMOTION::VEL_LAT, pi::HEXAMOTION::VEL_LAT, vel_variances_);
597 oa::setContinuousStateCovarianceAt(objectTemp, pi::HEXAMOTION::ACC_LON, pi::HEXAMOTION::ACC_LON, acc_variances_);
598 oa::setContinuousStateCovarianceAt(objectTemp, pi::HEXAMOTION::ACC_LAT, pi::HEXAMOTION::ACC_LAT, acc_variances_);
599 oa::setContinuousStateCovarianceAt(objectTemp, pi::HEXAMOTION::ROLL, pi::HEXAMOTION::ROLL, angle_variances_);
600 oa::setContinuousStateCovarianceAt(objectTemp, pi::HEXAMOTION::PITCH, pi::HEXAMOTION::PITCH, angle_variances_);
601 oa::setContinuousStateCovarianceAt(objectTemp, pi::HEXAMOTION::YAW, pi::HEXAMOTION::YAW, angle_variances_);
602 oa::setContinuousStateCovarianceAt(objectTemp, pi::HEXAMOTION::ROLL_RATE, pi::HEXAMOTION::ROLL_RATE, angle_rate_variances_);
603 oa::setContinuousStateCovarianceAt(objectTemp, pi::HEXAMOTION::PITCH_RATE, pi::HEXAMOTION::PITCH_RATE, angle_rate_variances_);
604 oa::setContinuousStateCovarianceAt(objectTemp, pi::HEXAMOTION::YAW_RATE, pi::HEXAMOTION::YAW_RATE, angle_rate_variances_);
605 oa::setContinuousStateCovarianceAt(objectTemp, pi::HEXAMOTION::LENGTH, pi::HEXAMOTION::LENGTH, pos_variances_);
606 oa::setContinuousStateCovarianceAt(objectTemp, pi::HEXAMOTION::WIDTH, pi::HEXAMOTION::WIDTH, pos_variances_);
607 oa::setContinuousStateCovarianceAt(objectTemp, pi::HEXAMOTION::HEIGHT, pi::HEXAMOTION::HEIGHT, pos_variances_);
608
609 // global object list: Sensor ID 0
610 objectTemp.state.sensor_id.push_back(0);
611
612 objectTemp.state.classifications.resize(1);
613 objectTemp.state.classifications[0].probability = 1.0; // always 1.0 as source is CARLA
614 switch (static_cast<int>(msg->objects[i].classification)) {
615 case dom::Object::CLASSIFICATION_PEDESTRIAN:
616 objectTemp.state.classifications[0].type = pi::ObjectClassification::PEDESTRIAN;
617 break;
618 case dom::Object::CLASSIFICATION_BIKE:
619 objectTemp.state.classifications[0].type = pi::ObjectClassification::BICYCLE;
620 break;
621 case dom::Object::CLASSIFICATION_MOTORCYCLE:
622 objectTemp.state.classifications[0].type = pi::ObjectClassification::MOTORCYCLE;
623 break;
624 case dom::Object::CLASSIFICATION_CAR:
625 objectTemp.state.classifications[0].type = pi::ObjectClassification::CAR;
626 break;
627 case dom::Object::CLASSIFICATION_TRUCK:
628 objectTemp.state.classifications[0].type = pi::ObjectClassification::UTILITY;
629 break;
630 default:
631 objectTemp.state.classifications[0].type = pi::ObjectClassification::UNCLASSIFIED;
632 break;
633 }
634
635 // set z for vehicles to center
636 if (objectTemp.state.classifications[0].type != pi::ObjectClassification::PEDESTRIAN) {
637 oa::setZ(objectTemp, msg->objects[i].pose.position.z + msg->objects[i].shape.dimensions[2] / 2.0);
638 }
639
640 objectTemp.state.reference_point.value = pi::ObjectReferencePoint::GEOMETRIC_CENTER;
641
642 msg_object_list_.objects.push_back(objectTemp);
643 }
644 return msg_object_list_;
645}

663bool CarlaConverter::transformFrame(const pi::ObjectList& msg_object_list,
664 pi::ObjectList& msg_object_list_transformed,
665 std::string target_frame) {
666 if (msg_object_list.objects.empty()) return false;
667
668 try {
669 while (true) {
670 if (tf2_buffer_->_frameExists(target_frame)) {
671 msg_object_list_transformed = tf2_buffer_->transform(msg_object_list, target_frame, tf2::durationFromSec(0.1));
672 return true;
673 }
674
675 // if target_frame frame does not exist, try to transform to sub_frame
676 size_t pos = target_frame.rfind("/");
677 if (pos != std::string::npos) {
678 target_frame = target_frame.substr(0, pos);
679 } else {
680 return false;
681 }
682 }
683 } catch (tf2::TransformException& e) {
684 ROS_LOG_STREAM(WARN, "Transformation from '" << target_frame << "' or its subframes is not available");
685 ROS_LOG_STREAM(WARN, e.what());
686 return false;
687 }
688}
